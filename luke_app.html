// React Component for Chapter Card
function ChapterCard({ chapter, verses, summary }) {
  return (
    <div className="bg-white p-6 m-4 rounded-lg shadow-md hover:shadow-lg transition">
      <h2 className="text-xl font-bold text-blue-600">Chapter {chapter}</h2>
      <p className="text-gray-600 mt-2">{summary}</p>
      <p className="text-gray-500 mt-2 text-sm">{verses}</p>
      <button
        className="mt-4 bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600"
        onClick={() => alert(`Reading Luke Chapter ${chapter}`)}
      >
        Read Chapter
      </button>
      <button
        className="mt-2 ml-2 bg-green-500 text-white px-4 py-2 rounded hover:bg-green-600"
        onClick={() => speak(`Chapter ${chapter}: ${summary}`)}
      >
        Listen
      </button>
      <button
        className="mt-2 ml-2 bg-gray-500 text-white px-4 py-2 rounded hover:bg-gray-600"
        onClick={() => shareChapter(chapter)}
      >
        Share
      </button>
    </div>
  );
}

// Main App Component
function App() {
  const [searchTerm, setSearchTerm] = React.useState('');

  const filteredChapters = lukeData.chapters.filter(ch =>
    ch.summary.toLowerCase().includes(searchTerm.toLowerCase()) ||
    ch.verses.toLowerCase().includes(searchTerm.toLowerCase())
  );

  return (
    <div className="container mx-auto p-4">
      <h1 className="text-3xl font-bold text-center text-gray-800 mb-6">The Book of Luke</h1>
      <div className="mb-6">
        <input
          type="text"
          placeholder="Search chapters..."
          className="w-full p-2 border rounded"
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
        />
      </div>
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {filteredChapters.map(ch => (
          <ChapterCard
            key={ch.chapter}
            chapter={ch.chapter}
            verses={ch.verses}
            summary={ch.summary}
          />
        ))}
      </div>
    </div>
  );
}

// Speech Synthesis for audio narration
function speak(text) {
  const utterance = new SpeechSynthesisUtterance(text);
  window.speechSynthesis.speak(utterance);
}

// Share functionality
function shareChapter(chapter) {
  if (navigator.share) {
    navigator.share({
      title: `Book of Luke - Chapter ${chapter}`,
      text: `Explore Chapter ${chapter} of the Book of Luke!`,
      url: window.location.href
    }).catch(err => console.error('Share failed:', err));
  } else {
    alert('Sharing not supported on this device. Copy the link to share!');
  }
}

// Render the App
const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
